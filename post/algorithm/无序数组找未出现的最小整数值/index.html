在无序数组中找出最小的正整数

Dec 01, 2020

  
   /**
     * return the min number
     * @param arr int整型一维数组 the array
     * @return int整型
     */
    public int minNumberdisappered (int[] arr) {
        // write code here
        Arrays.sort(arr);
        int min = 1;
        for(int i = 0;i<arr.length;i++){
            if(arr[i]==min)
                min++; 
        }
        return min;
    }
